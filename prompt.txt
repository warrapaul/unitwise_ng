@Tag(name = "Authentication", description = "Endpoints for user authentication, registration, password management, and session handling.")
@CrossOrigin(origins = "*", maxAge = 3600)
@RestController
@RequestMapping("/v1/auth")
@RequiredArgsConstructor
@Slf4j
public class AuthController {
    private final AuthService authService;

    @Operation(summary = "Login with email and password", description = "Authenticates user with email and password credentials")
    @PostMapping("/login")
    public ResponseEntity<ApiResponse<JwtResponseDto>> authenticateUser(@Valid @RequestBody LoginRequestDto loginRequest) {
        JwtResponseDto response = authService.loginUser(loginRequest);
        return ResponseEntity.ok(ApiResponse.success("Login successful", response));
    }

    /**
     * Check what login method user needs
     * Frontend calls this first to determine UI flow - show password or otp fields
     * if is temporary prompt user to complete registration profile
     */
    @Operation(summary = "Check login method", description = "Determines whether user should use password or OTP login based on account status")
    @PostMapping("/check-login-method")
    public ResponseEntity<ApiResponse<LoginMethodResponse>> checkLoginMethod(
            @Valid @RequestBody CheckLoginMethodRequest request) {

        LoginMethodResponse response = authService.checkLoginMethod(request.getPhoneNumber());

        return ResponseEntity.ok(ApiResponse.success("Login method checked successfully", response));
    }



    /**
     * Request OTP for login (separate from registration OTP)
     */
    @Operation(summary = "Request OTP for login", description = "Requests one-time password for login authentication")
    @PostMapping("/login-otp/request")
    public ResponseEntity<ApiResponse<OtpRequestResponse>> requestLoginOtp(
            @Valid @RequestBody OtpVerificationDtos.RequestOtpDto request) {
        OtpRequestResponse result = authService.requestLoginOtp(request);
        return ResponseEntity.ok(ApiResponse.success(result));
    }


    /**
     * Login using OTP (passwordless authentication)
     */
    @Operation(summary = "Login with OTP", description = "Authenticates user using one-time password")
    @PostMapping("/login-otp/confirm")
    public ResponseEntity<ApiResponse<JwtResponseDto>> loginWithOtp(
            @Valid @RequestBody OtpVerificationDtos.VerifyOtpDto request) {
        JwtResponseDto response = authService.verifyLoginOtp(request);
        return ResponseEntity.ok(ApiResponse.success("Login successful", response));
    }



    /**
     *  Request OTP for phone verification before signup
     */
    @Operation(summary = "Request signup phone verification OTP", description = "Requests OTP for verifying phone number during user registration")
    @PostMapping("/signup/verify-phone/request")
    public ResponseEntity<ApiResponse<OtpRequestResponse>> requestSignupVerification(
            @Valid @RequestBody OtpVerificationDtos.RequestOtpDto request) {
        OtpRequestResponse result = authService.requestSignupPhoneVerification(request);
        return ResponseEntity.ok(ApiResponse.success("Verification OTP sent", result));
    }

    //TODO: ensure all saved numbers in db have 254 prefix


    /**
     * Verify OTP for phone number before allowing signup
     */
    @Operation(summary = "Confirm signup phone verification", description = "Verifies OTP for phone number during user registration")
    @PostMapping("/signup/verify-phone/confirm")
    public ResponseEntity<ApiResponse<VerifyOtpResponse>> confirmSignupVerification(
            @Valid @RequestBody OtpVerificationDtos.VerifyOtpDto request) {
        VerifyOtpResponse result = authService.confirmSignupVerification(request);
        return ResponseEntity.ok(ApiResponse.success("Phone verified successfully", result));
    }


    @Operation(summary = "Register new user", description = "Registers a new user after phone verification")
    @PostMapping("/signup")
    public ResponseEntity<ApiResponse<JwtResponseDto>> registerUser(

        JwtResponseDto response = authService.registerUser(registerRequest);
        return ResponseEntity.status(HttpStatus.CREATED)
                .body(ApiResponse.success("User registered successfully", response));

    @Operation(summary = "Refresh JWT token", description = "Refreshes authentication token using refresh token")

    /**
     * Requests OTP for phone verification during user signup
     */
    @Override
    public OtpRequestResponse requestSignupPhoneVerification(OtpVerificationDtos.RequestOtpDto request) {
        String cleanPhone = PhoneNumberUtil.clean(request.getPhoneNumber());
        if (userRepository.findByPhoneNumber(cleanPhone).isPresent()) {
            throw new BadRequestException("Phone number is already registered proceed to login");
        }
        return otpService.requestOtp(cleanPhone, OtpType.SIGNUP_VERIFICATION);
    }
    
    
    /**
    * Confirms phone verification OTP during signup process
    */
    @Override
    public VerifyOtpResponse confirmSignupVerification(OtpVerificationDtos.VerifyOtpDto request) {        
        String cleanPhone = PhoneNumberUtil.clean(request.getPhoneNumber());        
        VerifyOtpResponse otpResult = verifyOtpOrThrow(cleanPhone, request.getOtp());
        return otpResult;
    }
    
    
    
    /**
     * Registers a new user after successful phone verification
     * Creates user account and returns JWT tokens for immediate login
     */
    @Override
    @Transactional
    public JwtResponseDto registerUser(UserDtos.RegisterRequest request) {
    	String cleanPhone = PhoneNumberUtil.clean(request.getPhoneNumber());

    	// Convert RegisterRequest to CreateRequest
    	UserDtos.CreateRequest createRequest = UserDtos.CreateRequest.builder()
    		    .firstName(request.getFirstName())
    		    .middleName(request.getMiddleName())
    		    .lastName(request.getLastName())
    		    .email(request.getEmail())
    		    .password(request.getPassword())
    		    .phoneNumber(cleanPhone)
    		    .nationalIdNumber(request.getNationalIdNumber())
    		    .build();

        
        // Verify phone was actually verified by checking OTP service
        if (!otpService.isPhoneVerified(cleanPhone)) {
            throw new BadRequestException("Phone number verification required or verification token expired");
        }
        
        
        UserDtos.DetailResponse newUser = userService.userSignUp(createRequest);

        Authentication authentication = authenticationManager.authenticate(
                new UsernamePasswordAuthenticationToken(
                        newUser.getEmail(),
                        request.getPassword()));

        SecurityContextHolder.getContext().setAuthentication(authentication);
        UserPrincipal principal = (UserPrincipal) authentication.getPrincipal();
        User user = principal.getUser();

        return buildJwtResponse(user);
    }

    

    /**
     * Authenticates user with email and password credentials
     */
    @Override
    @Transactional
    public JwtResponseDto loginUser(LoginRequestDto loginRequest) {
        Authentication authentication = authenticationManager.authenticate(
                new UsernamePasswordAuthenticationToken(
                        loginRequest.getEmail(),
                        loginRequest.getPassword()));

        SecurityContextHolder.getContext().setAuthentication(authentication);
        UserPrincipal principal = (UserPrincipal) authentication.getPrincipal();
        User user = principal.getUser();

        return handlePasswordResetOrBuildJwt(user);
    }
    
    /**
     * Requests OTP for login when user has temporary account or password login not available
     * Full accounts: Password required for login
     */
	@Override
	public OtpRequestResponse requestLoginOtp(RequestOtpDto request) {
        String cleanPhone = PhoneNumberUtil.clean(request.getPhoneNumber());

        boolean otpAvailable = checkLoginMethod(cleanPhone).isRequiresPassword();
    
        if(!otpAvailable) {
            throw new BadRequestException("Account only supports login by password. Proceed to password login or reset password");
        }
		return otpService.requestOtp(request.getPhoneNumber(), OtpType.LOGIN);		
	}
	

    /*
     * Determines the appropriate login method for a user based on account status
     * Temporary accounts (no password) use OTP login, full accounts use password
     */
    @Override
    @Transactional
    public LoginMethodResponse checkLoginMethod(String phoneNumber) {
        String cleanPhone = PhoneNumberUtil.clean(phoneNumber);

        User user = userRepository.findByPhoneNumber(cleanPhone)
                .orElseThrow(() -> new ResourceNotFoundException("User", "phone", phoneNumber));

        boolean isTemporary = user.getIsTemporary() || user.getPassword() == null;

        String message = user.getIsTemporary()
                ? "OTP verification required to complete registration"
                : user.getPassword() == null
                    ? "OTP login available"
                    : "Password login required";

        return LoginMethodResponse.builder()
                .requiresPassword(!isTemporary)
                .temporary(isTemporary)
                .build();
    }

	public JwtResponseDto verifyLoginOtp(VerifyOtpDto request) {
    	String cleanPhone = PhoneNumberUtil.clean(request.getPhoneNumber());

    	VerifyOtpResponse result = otpService.verifyOtp(cleanPhone, request.getOtp());
        if (!result.isSuccess()) {
            throw new BadRequestException(result.getMessage());
        }

	    User user = userRepository.findByPhoneNumber(cleanPhone)
	            .orElseThrow(() -> new ResourceNotFoundException("User", "phone number", cleanPhone));
	        
        if (!isUserActiveAndNotLocked(user)) {
            throw new BadRequestException("User account is locked or inactive");
        }
        
        // Update last login time
        if (user.getUserAccount() != null) {
            user.getUserAccount().setLastLogin(LocalDateTime.now());
            userRepository.save(user);
        }
        
        setAuthenticationContext(user);
        return handlePasswordResetOrBuildJwt(user);
    }

//=============================================================================

    @Override
    @Transactional
    public JwtResponseDto refreshToken(TokenRefreshRequest request) {
        RefreshToken refreshToken = refreshTokenService.findByToken(request.getRefreshToken())
                .map(refreshTokenService::verifyExpiration)
                .orElseThrow(() -> new TokenRefreshException(
                        request.getRefreshToken(),
                        "Refresh token is invalid or expired"));

        User user = refreshToken.getUser();
        setAuthenticationContext(user);

        JwtResponseDto newAuth = buildJwtResponse(user);

        // Delete old refresh token
        refreshTokenService.deleteByToken(refreshToken.getToken());
        return newAuth;
    }

    @Override
    public String logoutUser() {
        Authentication auth = SecurityContextHolder.getContext().getAuthentication();
        if (auth != null) {
            UserPrincipal principal = (UserPrincipal) auth.getPrincipal();
            User user = principal.getUser();
            refreshTokenService.deleteByUserId(user.getId());
        }
        return "Log out successful!";
    }
    

    //---------------------- BUILD JWT RESPONSE --------------------
    private JwtResponseDto buildJwtResponse(User user) {
        String jwt = jwtTokenUtil.generateTokenFromUserId(user.getId());// generateTokenFromUsername(user.getEmail());
        RefreshToken refreshToken = refreshTokenService.createRefreshToken(user);

        return JwtResponseDto.builder()
                .accessToken(jwt)
                .refreshToken(refreshToken.getToken())
                .build();
    }    
    
     /**
     * Handles password reset requirement check and returns appropriate JWT response
     */
    private JwtResponseDto handlePasswordResetOrBuildJwt(User user) {
        if (user.getUserAccount() != null && user.getUserAccount().isPasswordResetRequired()) {
            String tempJwt = jwtTokenUtil.generatePasswordChangeToken(user.getId());
            return JwtResponseDto.builder()
                    .accessToken(tempJwt)
                    .refreshToken(null)
                    .passwordResetRequired(true)
                     // .id(user.getId())
                     // .expiresIn(jwtTokenUtil.getJwtExpirationSec())
                    .build();
        }

        return  buildJwtResponse(user);      

    }

    /**
     * Completes registration for temporary account by setting password
     * Converts temporary account to full account with proper credentials
     */
    @Override
    @Transactional
    public JwtResponseDto setPassword(PasswordSetRequestDto request) {
        String cleanPhone = PhoneNumberUtil.clean(request.getPhoneNumber());
        
        if (!otpService.isVerificationTokenValid(cleanPhone, request.getVerificationToken())) {
            throw new BadRequestException("Invalid or expired verification. Please verify OTP again.");
        }
        
        User user = userRepository.findByPhoneNumber(cleanPhone)
                .orElseThrow(() -> new ResourceNotFoundException("User", "phone", cleanPhone));
        
        boolean isTemporary = user.getIsTemporary() || user.getPassword() == null;
        if (!isTemporary) {
            throw new BadRequestException("Account has already been completed");
        }
        
        validateAndSetNewPassword(user, request.getPassword());
        
        //Complete the registration
        // user.setNationalIdNumber(request.getNationalIdNumber()); // optional complete profile
        user.setIsTemporary(false);
        
        // Update account status
        if (user.getUserAccount() != null) {
            user.getUserAccount().setPhoneVerified(true);
            user.getUserAccount().setLastLogin(LocalDateTime.now());
        }
        
        userRepository.save(user);
        
        otpService.invalidateVerificationToken(cleanPhone);
        
        log.info("Temporary account registration completed: Phone={}, UserID={}", cleanPhone, user.getId());
        
        setAuthenticationContext(user);
        return handlePasswordResetOrBuildJwt(user);
    }
    
    
    
    @Override
    @Transactional
    public String userInitiatePasswordReset(PasswordResetDtos.UserInitiateRequest request) {
    	String email = request.getEmail();       // ALWAYS return success to prevent email enumeration - only send email if user exists and is active        
        Optional<User> userOpt = userRepository.findByEmail(email);
        
        if (userOpt.isPresent()) {
            User user = userOpt.get();
            
            if (isUserActiveAndNotLocked(user)) {                
            	String token = createPasswordResetToken(user, 1); // 1 hour expiry
                
                // Send reset email
                emailService.sendPasswordResetEmail(user.getEmail(), token, user.getFirstName());
                
                log.info("Password reset initiated for user: {}", email);
            } else {
                log.warn("Password reset attempted for inactive/locked user: {}", email);
            }
        } else {
            log.warn("Password reset attempted for non-existent email: {}", email);
        }
        
        // ALWAYS return success - no information leakage
        return "If an account with that email exists, a password reset link has been sent.";
    }
    
    @Override
    @Transactional
    public String adminResetPassword(PasswordResetDtos.AdminInitiateRequest request) {
    	Long userId = request.getUserId();
    	User user =  userRepository.findById(userId)
                .orElseThrow(() -> new ResourceNotFoundException("User", "id", userId));

        // Set flag requiring password reset
        if (user.getUserAccount() != null) {
            user.getUserAccount().setPasswordResetRequired(true);
        }
        
        // Create reset token with 24 hour expiry for admin reset
        String token = createPasswordResetToken(user, 24);
        
        // Send admin-initiated reset email
        emailService.sendAdminPasswordResetEmail(user.getEmail(), token, user.getFirstName());
        
        log.info("Admin initiated password reset for user ID: {}", userId);
        return "Password reset initiated for user";
    }

    
    @Override
    @Transactional
    // Password reset tokens - used for password resets, not authentication
    public String setNewPasswordAfterResetRequest(PasswordResetDtos.ConfirmRequest request) {
    	String token = request.getToken();
    	String newPassword = request.getNewPassword();
        PasswordResetToken resetToken  = passwordResetTokenRepository.findByToken(token)
                .orElseThrow(() -> new ResourceNotFoundException("Password reset", "token", token));

        if (resetToken.getExpiryDate().isBefore(LocalDateTime.now())) {
        	passwordResetTokenRepository.delete(resetToken);
            throw new BadRequestException("Reset token has expired"); 
        }
        
        User user = resetToken.getUser();
        
        if (!isUserActiveAndNotLocked(user)) {
            throw new BadRequestException("User account is not active");
        }
        
        // Validate and set password
        validateAndSetNewPassword(user, newPassword);
        
        // Update user account
        resetUserAccountAfterPasswordChange(user);
        
        userRepository.save(user);
        
        // Delete the token (single use)
        passwordResetTokenRepository.delete(resetToken);
        
        log.info("Password reset completed for user: {}", user.getEmail());
        return "Password has been reset successfully. Proceed to login";
    }
    
    @Override
    @Transactional
    public JwtResponseDto changePassword(PasswordResetDtos.PasswordChangeRequestDto request, UserPrincipal principal) {
        User user = principal.getUser();

        // Verify current password
        if (!passwordEncoder.matches(request.getCurrentPassword(), user.getPassword())) {
         
//            // Increment failed attempts if user account exists
//            if (user.getUserAccount() != null) {
//                incrementFailedPasswordChangeAttempts(user);
//            }
            
            throw new BadRequestException("Current password is incorrect");
        }
        
        validateAndSetNewPassword(user, request.getNewPassword());
                
        try {
            // Update user account metadata
            resetUserAccountAfterPasswordChange(user);
            
            // Save user - will cascade to save account
            userRepository.save(user);
            
            // Invalidate existing refresh tokens for security
            refreshTokenService.deleteByUserId(user.getId());
            
            // Generate new JWT response
            JwtResponseDto jwtResponse = buildJwtResponse(user);
            
            
            // Optional: Send notification email
            // emailService.sendPasswordChangeNotification(user.getEmail(), user.getFirstName());
            
            return jwtResponse;
            
        } catch (Exception e) {
            log.error("Failed to change password for user {}: {}", user.getEmail(), e.getMessage());
            throw new BadRequestException("Failed to change password. Please try again.");
        }
    }

    /**
     * Verifies OTP and throws BadRequestException if verification fails
     * Reused across: confirmSignupVerification, verifyLoginOtpForTemporaryAccount, 
     * verifyLoginOtpForFullAccount, verifyLoginOtp
     */
    private VerifyOtpResponse verifyOtpOrThrow(String cleanPhone, String otp) {
        VerifyOtpResponse result = otpService.verifyOtp(cleanPhone, otp);
        if (!result.isSuccess()) {
            throw new BadRequestException(result.getMessage());
        }
        return result;
    }
    
    /**
     * Sets authentication context with user principal
     * Reused across: verifyLoginOtpForTemporaryAccount, completeTemporaryAccountRegistration,
     * verifyLoginOtpForFullAccount, verifyLoginOtp, refreshToken
     */
    private void setAuthenticationContext(User user) {
        UserPrincipal principal = new UserPrincipal(user);
        UsernamePasswordAuthenticationToken authentication = 
            new UsernamePasswordAuthenticationToken(principal, null, principal.getAuthorities());
        SecurityContextHolder.getContext().setAuthentication(authentication);
    }
    
   
    
    /**
     * Validates password strength and ensures it's different from current password
     */
    private void validateAndSetNewPassword(User user, String newPassword) {
        PasswordValidator.validatePasswordStrength(newPassword);
        
        // Ensure new password is different from current
        if (user.getPassword() != null && passwordEncoder.matches(newPassword, user.getPassword())) {
            throw new BadRequestException("New password must be different from current password");
        }
        
        user.setPassword(passwordEncoder.encode(newPassword));
    }
    
    /**
     * Resets user account flags after successful password change
     * Reused across: setNewPasswordAfterResetRequest, changePassword
     */
    private void resetUserAccountAfterPasswordChange(User user) {
        if (user.getUserAccount() != null) {
            UserAccount account = user.getUserAccount();
            account.setPasswordResetRequired(false);
            account.setPasswordChangedAt(LocalDateTime.now());
            account.setFailedLoginAttempts(0);
            account.setAccountLockedUntil(null);
            account.setIsLocked(false);
        }
    }
    
    /**
     * Creates a password reset token for the user
     * Reused across: userInitiatePasswordReset, adminResetPassword
     * @param user User to create token for
     * @param expiryHours Number of hours until token expires
     * @return The generated token string
     */
    private String createPasswordResetToken(User user, int expiryHours) {
        // Delete any existing tokens for this user (only one active token at a time)
        passwordResetTokenRepository.deleteByUser(user);
        
        // Generate new reset token
        String token = generateResetToken();
        PasswordResetToken resetToken = PasswordResetToken.builder()
            .token(token)
            .user(user)
            .expiryDate(LocalDateTime.now().plusHours(expiryHours))
            .build();
        passwordResetTokenRepository.save(resetToken);
        
        return token;
    }
    
    private boolean isUserActiveAndNotLocked(User user) {
        if (user.getUserAccount() == null) {
            return true; // If no account info, assume active
        }
        
        boolean isNotLocked = !user.getUserAccount().getIsLocked();
        boolean isNotTimeLocked = user.getUserAccount().getAccountLockedUntil() == null ||
                                user.getUserAccount().getAccountLockedUntil().isBefore(LocalDateTime.now());
        
        return isNotLocked && isNotTimeLocked;
    }
       

    private String generateResetToken() {
        // Generate a secure 64-character token
        return UUID.randomUUID().toString().replace("-", "") + 
               UUID.randomUUID().toString().replace("-", "");
    }
    
    @Scheduled(fixedRate = 1800000)
    @Transactional
    public void cleanupExpiredTokens() { // runs every 30 minutes
        int deletedCount = passwordResetTokenRepository.deleteByExpiryDateBefore(LocalDateTime.now());
        if (deletedCount > 0) {
            log.info("Cleaned up {} expired password reset tokens", deletedCount);
        }
    }
    
    // Unlock accounts that have passed their lockout time - runs every 5 minutes   
    @Scheduled(fixedRate = 300000)
    @Transactional
    public void unlockExpiredAccounts() {
        List<User> lockedUsers = userRepository.findByUserAccount_AccountLockedUntilBefore(LocalDateTime.now());
        for (User user : lockedUsers) {
            if (user.getUserAccount() != null) {
                user.getUserAccount().setAccountLockedUntil(null);
                user.getUserAccount().setFailedLoginAttempts(0);
                user.getUserAccount().setIsLocked(false);
                userRepository.save(user);
            }
        }
        if (!lockedUsers.isEmpty()) {
            log.info("Unlocked {} expired account locks", lockedUsers.size());
        }
    }
}

@Builder

public class UserSearchReq {

    @Size(max = 50, message = "First name must not exceed 50 characters")
    @Schema(description = "User's first name")
    private String firstName;

    @Size(max = 50, message = "Last name must not exceed 50 characters")
    @Schema(description = "User's last name")
    private String lastName;

    @Size(max = 100, message = "Email must not exceed 100 characters")
    @Schema(description = "User's email address")
    private String email;

    @Size(max = 20, message = "Phone number must not exceed 20 characters")
    @Schema(description = "User's phone number")
    private String phoneNumber;

    @Size(max = 50, message = "National ID must not exceed 50 characters")
    @Schema(description = "User's national ID number")
    private String nationalId;

    @Size(max = 50, message = "User ID must not exceed 50 characters")
    @Schema(description = "Unique user identifier")
    private String userUid;

    // Clean methods
    @JsonIgnore
    public String getCleanFirstName() {
        return StringUtils.hasText(firstName) ? firstName.trim() : null;
    }

    @JsonIgnore
    public String getCleanLastName() {
        return StringUtils.hasText(lastName) ? lastName.trim() : null;
    }

    @JsonIgnore
    public String getCleanEmail() {
        return StringUtils.hasText(email) ? email.trim() : null;
    }

    @JsonIgnore
    public String getCleanPhoneNumber() {
        return StringUtils.hasText(phoneNumber) ? phoneNumber.trim() : null;
    }

    @JsonIgnore
    public String getCleanNationalId() {
        return StringUtils.hasText(nationalId) ? nationalId.trim() : null;
    }

    @JsonIgnore
    public String getCleanUserUid() {
        return StringUtils.hasText(userUid) ? userUid.trim() : null;
    }

    // Check if any search criteria are present
    public boolean hasAnySearchCriteria() {
        return getCleanFirstName() != null ||
                getCleanLastName() != null ||
                getCleanEmail() != null ||
                getCleanPhoneNumber() != null ||
                getCleanNationalId() != null ||
                getCleanUserUid() != null;
    }

    public boolean isEmpty() {
        return !hasAnySearchCriteria();
    }
}

/**
 * Mapper for converting User entities to DTOs.
 * Centralizes user mapping logic to avoid duplication across services.
 */
@Component
@RequiredArgsConstructor
public class UserMapper {

    private final ModelMapper modelMapper;
    private final UserRepository userRepository;

    /**
     * Convert User entity to DetailResponse DTO
     */
    public UserDtos.DetailResponse toDetailResponse(User user) {
        if (user == null) return null;

        return UserDtos.DetailResponse.builder()
                .id(user.getId())
                .firstName(user.getFirstName())
                .middleName(user.getMiddleName())
                .lastName(user.getLastName())
                .email(user.getEmail())
                .userUid(user.getUserUid())
                .phoneNumber(user.getPhoneNumber())
                .nationalIdNumber(user.getNationalIdNumber())
                .status(user.getStatus())
                .isActive(user.getUserAccount() != null ? !user.getUserAccount().getIsLocked() : true)
                .emailVerified(user.getUserAccount() != null ? user.getUserAccount().isEmailVerified() : false)
                .createdAt(user.getCreatedAt())
                .lastLogin(user.getUserAccount() != null ? user.getUserAccount().getLastLogin() : null)
                .updatedAt(user.getUpdatedAt())
                .userProfile(toUserProfileResponse(user.getUserProfile()))
                .userAccount(toUserAccountResponse(user.getUserAccount()))
                .roles(user.getRoles() != null ? user.getRoles().stream()
                        .filter(role -> role.getEnabled())
                        .map(role -> modelMapper.map(role, RoleDtos.ResponseDto.class))
                        .collect(Collectors.toSet()) : Collections.emptySet())
                .build();
    }

    /**
     * Convert User entity to PreviewResponse DTO
     */
    public UserDtos.PreviewResponse toPreviewResponse(User user) {
        if (user == null) return null;

        return UserDtos.PreviewResponse.builder()
                .id(user.getId())
                .firstName(user.getFirstName())
                .middleName(user.getMiddleName())
                .lastName(user.getLastName())
                .email(user.getEmail())
                .userUid(user.getUserUid())
                .phoneNumber(user.getPhoneNumber())
                .nationalIdNumber(user.getNationalIdNumber())
                .isActive(user.getUserAccount() != null ? !user.getUserAccount().getIsLocked() : true)
                .emailVerified(user.getUserAccount() != null ? user.getUserAccount().isEmailVerified() : false)
                .createdAt(user.getCreatedAt())
                .lastLogin(user.getUserAccount() != null ? user.getUserAccount().getLastLogin() : null)
                .profileImageUrl(user.getUserProfile() != null ? user.getUserProfile().getProfileImageUrl() : null)
                .build();
    }

    /**
     * Convert UserProfile to DTO
     */
    public UserDtos.UserProfileResponseDto toUserProfileResponse(UserProfile profile) {
        if (profile == null) return null;

        return UserDtos.UserProfileResponseDto.builder()
                .id(profile.getId())
                .profileImageUrl(profile.getProfileImageUrl())
                .idPicFront(profile.getIdPicFront())
                .idPicBack(profile.getIdPicBack())
                .phoneNumberSecondary(profile.getPhoneNumberSecondary())
                .dateOfBirth(profile.getDateOfBirth())
                .gender(profile.getGender())
                .createdAt(profile.getCreatedAt())
                .updatedAt(profile.getUpdatedAt())
                .build();
    }

    /**
     * Convert UserAccount to DTO
     */
    public UserDtos.UserAccountResponseDto toUserAccountResponse(UserAccount account) {
        if (account == null) return null;

        return UserDtos.UserAccountResponseDto.builder()
                .id(account.getId())
                .emailVerified(account.isEmailVerified())
                .phoneVerified(account.isPhoneVerified())
                .lastLogin(account.getLastLogin())
                .isLocked(account.getIsLocked())
                .lockedUntil(account.getLockedUntil())
                .failedLoginAttempts(account.getFailedLoginAttempts())
                .passwordResetRequired(account.isPasswordResetRequired())
                .passwordChangedAt(account.getPasswordChangedAt())
                .createdAt(account.getCreatedAt())
                .updatedAt(account.getUpdatedAt())
                .build();
    }

    /**
     * Get full name of a user by ID
     * Used by other mappers (e.g., TenantMapper) for verifier names
     */
    public String getFullName(Long userId) {
        if (userId == null) return null;

        return userRepository.findById(userId)
                .map(user -> user.getFirstName() + " " + user.getLastName())
                .orElse(null);
    }
}

    public DetailResponse userSignUp(CreateRequest registerRequestDto) {
    	String cleanPhone = PhoneNumberUtil.clean(registerRequestDto.getPhoneNumber());

        if (userRepository.existsByEmail(registerRequestDto.getEmail())) {
            throw new ResourceAlreadyExistsException("User", "email", registerRequestDto.getEmail());
        }

        if (userRepository.existsByPhoneNumber(cleanPhone)) {
            throw new ResourceAlreadyExistsException("User", "phone", registerRequestDto.getPhoneNumber());
        }
                      
        Roles userRole = roleRepository.findByName("GUEST")
                .orElseThrow(() -> new RuntimeException("Default User role not found"));
                
        PasswordValidator.validatePasswordStrength(registerRequestDto.getPassword());

        //generate a 9 alphanumeric user uid - check if it exists in db first else regenerate
        String userUid = generateUserUid();

        // MULT-USER-ROLE:   USER WITH MANY ROLES
        User user = User.builder()
            .firstName(registerRequestDto.getFirstName())
            .middleName(registerRequestDto.getMiddleName())
            .lastName(registerRequestDto.getLastName())
            .email(registerRequestDto.getEmail())
            .phoneNumber(cleanPhone)
            .nationalIdNumber(registerRequestDto.getNationalIdNumber())
            .password(passwordEncoder.encode(registerRequestDto.getPassword()))
            .userUid(userUid)
            .build();

        // assign the default USER role using the many-to-many relationship
        user.getRoles().add(userRole);
                       
        UserProfile profile = UserProfile.builder()
                .user(user)
                .build();

        UserAccount account = UserAccount.builder()
                .emailVerified(false)
                .phoneVerified(false)
                .isLocked(false)
                .failedLoginAttempts(0)
                .user(user)
                .build();

        user.setUserProfile(profile);
        user.setUserAccount(account);

        User savedUser = userRepository.save(user);
        return userMapper.toDetailResponse(savedUser);
    }

    /**
     * handle all user creation logics - landlord, admins, tenant accs by landlord
     * creator specifies role ids to be assigned
     * TODO: ensure creator has right permissions to assign those roles

     * a temporary password sent via email/sms
     * initial password is sent to their email/phone for first-time login verification
     */
    @Override
    @Transactional
    public User createUserWithTemporaryPassword(CreateRequest request, Long createdByUserId) {
        String cleanPhone = PhoneNumberUtil.clean(request.getPhoneNumber());

        if (userRepository.existsByEmail(request.getEmail())) {
            throw new ResourceAlreadyExistsException("User", "email", request.getEmail());
        }
        if (userRepository.existsByPhoneNumber(cleanPhone)) {
            throw new ResourceAlreadyExistsException("User", "phone", request.getPhoneNumber());
        }

        String password = generateSecurePassword();
        String userUid = generateUserUid();

        User user = User.builder()
                .firstName(request.getFirstName())
                .middleName(request.getMiddleName())
                .lastName(request.getLastName())
                .email(request.getEmail())
                .phoneNumber(cleanPhone)
                .nationalIdNumber(request.getNationalIdNumber())
                .password(passwordEncoder.encode(password))
                .userUid(userUid)
                .build();

        // Always start with GUEST — roleService will diff and apply correct roles below
        Roles guestRole = roleRepository.findByNameAndRoleScope("GUEST", Roles.RoleScope.SYSTEM)
                .orElseThrow(() -> new ResourceNotFoundException("Role", "name", "GUEST"));
        user.getRoles().add(guestRole);

        user.setUserProfile(UserProfile.builder().user(user).build());
        user.setUserAccount(UserAccount.builder()
                .emailVerified(false)
                .phoneVerified(false)
                .isLocked(false)
                .failedLoginAttempts(0)
                .passwordResetRequired(true)
                .user(user)
                .build());

        User savedUser = userRepository.save(user);

        // If caller specified roles, delegate to roleService — reuses all guards and side effects
        if (request.getRoleIds() != null && !request.getRoleIds().isEmpty()) {
            roleService.updateUserRoles(
                    savedUser.getId(),
                    RoleDtos.UpdateUserRolesRequest.builder().roleIds(request.getRoleIds()).build(),
                    createdByUserId
            );
        }

        emailService.sendTemporaryCredentials(savedUser.getEmail(), password, savedUser.getFirstName());
        return userRepository.findById(savedUser.getId()).orElseThrow();
    }

    @Override
    @Transactional(readOnly = true)
    public DetailResponse getUserProfile(Long userId) {
        User user = userRepository.findById(userId)
                .orElseThrow(() -> new ResourceNotFoundException("User", "id", userId));

        DetailResponse userProfile = userMapper.toDetailResponse(user);

        // ── 1. System roles (from user_roles) ──────────────────────────────────
        Set<RoleDtos.ResponseDto> roleDtos = user.getRoles().stream()
                .filter(Roles::getEnabled)
                .map(role -> {
                    RoleDtos.ResponseDto roleDto = modelMapper.map(role, RoleDtos.ResponseDto.class);
                    Set<PermissionDtos.Response> permissionDtos = role.getPermissions().stream()
                            .map(p -> modelMapper.map(p, PermissionDtos.Response.class))
                            .collect(Collectors.toSet());
                    roleDto.setPermissions(permissionDtos);
                    return roleDto;
                })
                .collect(Collectors.toSet());

        // ── 2. Agency roles (from agency_admins) — merge in ───────────────────
        List<AgencyAdmin> agencyAdminRecords = agencyAdminRepository.findAgenciesWhereUserIsAdmin(userId);
        agencyAdminRecords.stream()
                .map(AgencyAdmin::getRole)
                .filter(role -> role != null && role.getEnabled())
                .map(role -> {
                    RoleDtos.ResponseDto roleDto = modelMapper.map(role, RoleDtos.ResponseDto.class);
                    Set<PermissionDtos.Response> permissionDtos = role.getPermissions().stream()
                            .map(p -> modelMapper.map(p, PermissionDtos.Response.class))
                            .collect(Collectors.toSet());
                    roleDto.setPermissions(permissionDtos);
                    return roleDto;
                })
                .forEach(roleDtos::add); // Set deduplicates by role id if equals/hashCode is on id

        userProfile.setRoles(roleDtos);
        return userProfile;
    }

    @Override
    @Transactional
    public DetailResponse updateUser(Long userId, UpdateRequest updateDto) {
        User user = userRepository.findById(userId)
                .orElseThrow(() -> new ResourceNotFoundException("User", "id", userId));

        if (updateDto.getEmail() != null && !updateDto.getEmail().equals(user.getEmail())) {
            if (userRepository.existsByEmail(updateDto.getEmail()))
                throw new ResourceAlreadyExistsException("User", "email", updateDto.getEmail());
            user.setEmail(updateDto.getEmail());
        }

        if (updateDto.getPhoneNumber() != null && !updateDto.getPhoneNumber().equals(user.getPhoneNumber())) {
            String clean = PhoneNumberUtil.clean(updateDto.getPhoneNumber());
            if (userRepository.existsByPhoneNumber(clean))
                throw new ResourceAlreadyExistsException("User", "phone", updateDto.getPhoneNumber());
            user.setPhoneNumber(clean);
        }

        if (updateDto.getFirstName() != null)      user.setFirstName(updateDto.getFirstName());
        if (updateDto.getMiddleName() != null)     user.setMiddleName(updateDto.getMiddleName());
        if (updateDto.getLastName() != null)       user.setLastName(updateDto.getLastName());
        if (updateDto.getPassword() != null)       user.setPassword(passwordEncoder.encode(updateDto.getPassword()));
        if (updateDto.getNationalIdNumber() != null) user.setNationalIdNumber(updateDto.getNationalIdNumber());

        UserProfile profile = user.getUserProfile();
        if (profile != null) {
            if (updateDto.getPhoneNumberSecondary() != null) profile.setPhoneNumberSecondary(updateDto.getPhoneNumberSecondary());
            if (updateDto.getProfileImageUrl() != null)      profile.setProfileImageUrl(updateDto.getProfileImageUrl());
            if (updateDto.getIdPicFront() != null)           profile.setIdPicFront(updateDto.getIdPicFront());
            if (updateDto.getIdPicBack() != null)            profile.setIdPicBack(updateDto.getIdPicBack());
            if (updateDto.getDateOfBirth() != null)          profile.setDateOfBirth(updateDto.getDateOfBirth());
            if (updateDto.getGender() != null)               profile.setGender(updateDto.getGender());
        }

        User saved = userRepository.save(user);
        return userMapper.toDetailResponse(saved);
    }

    @Override
    @Transactional
    public DetailResponse adminUpdateUser(Long userId, AdminUpdateRequest dto) {
        User user = this.findById(userId);

        // User core
        if (dto.getFirstName() != null)        user.setFirstName(dto.getFirstName());
        if (dto.getMiddleName() != null)       user.setMiddleName(dto.getMiddleName());
        if (dto.getLastName() != null)         user.setLastName(dto.getLastName());
        if (dto.getNationalIdNumber() != null) user.setNationalIdNumber(dto.getNationalIdNumber());
        if (dto.getStatus() != null)           user.setStatus(dto.getStatus());

        if (dto.getEmail() != null && !dto.getEmail().equals(user.getEmail())) {
            if (userRepository.existsByEmail(dto.getEmail()))
                throw new ResourceAlreadyExistsException("User", "email", dto.getEmail());
            user.setEmail(dto.getEmail());
        }
        if (dto.getPhoneNumber() != null && !dto.getPhoneNumber().equals(user.getPhoneNumber())) {
            String clean = PhoneNumberUtil.clean(dto.getPhoneNumber());
            if (userRepository.existsByPhoneNumber(clean))
                throw new ResourceAlreadyExistsException("User", "phone", dto.getPhoneNumber());
            user.setPhoneNumber(clean);
        }

        // UserAccount
        UserAccount account = user.getUserAccount();
        if (account != null) {
            if (dto.getIsLocked() != null)              account.setIsLocked(dto.getIsLocked());
            if (dto.getLockedUntil() != null)           account.setLockedUntil(dto.getLockedUntil());
            if (dto.getPasswordResetRequired() != null) account.setPasswordResetRequired(dto.getPasswordResetRequired());
            if (dto.getEmailVerified() != null)         account.setEmailVerified(dto.getEmailVerified());
            if (dto.getPhoneVerified() != null)         account.setPhoneVerified(dto.getPhoneVerified());
            if (Boolean.TRUE.equals(dto.getResetFailedLoginAttempts())) {
                account.setFailedLoginAttempts(0);
                account.setIsLocked(false);
                account.setLockedUntil(null);
                account.setAccountLockedUntil(null);
            }
        }

        // UserProfile
        UserProfile profile = user.getUserProfile();
        if (profile != null) {
            if (dto.getPhoneNumberSecondary() != null) profile.setPhoneNumberSecondary(dto.getPhoneNumberSecondary());
            if (dto.getProfileImageUrl() != null)      profile.setProfileImageUrl(dto.getProfileImageUrl());
            if (dto.getIdPicFront() != null)           profile.setIdPicFront(dto.getIdPicFront());
            if (dto.getIdPicBack() != null)            profile.setIdPicBack(dto.getIdPicBack());
            if (dto.getDateOfBirth() != null)          profile.setDateOfBirth(dto.getDateOfBirth());
            if (dto.getGender() != null)               profile.setGender(dto.getGender());
        }

        User saved = userRepository.save(user);
        appLogger.info("Admin updated user: ID={}, by admin", saved.getId());
        return userMapper.toDetailResponse(saved);
    }

    // Helper methods for password and UID generation (keep these as they're business logic)
    private String generateSecurePassword() {
        String uppercase = "ABCDEFGHJKLMNPQRSTUVWXYZ";
        String lowercase = "abcdefghijkmnopqrstuvwxyz";
        String digits = "23456789";
        String symbols = "!@#$%^&*";

        String allChars = uppercase + lowercase + digits;
        // String allChars = uppercase + lowercase + digits + symbols;

        SecureRandom random = new SecureRandom();
        StringBuilder password = new StringBuilder();
        int length = 6;

        for (int i = 0; i < length; i++) {
            password.append(allChars.charAt(random.nextInt(allChars.length())));
        }

        return password.toString();
    }

    private String generateUserUid() {
        String allowedCharacters = "23456789ABCDEFGHJKMNPQRSTUVWXYZ";
        SecureRandom random = new SecureRandom();

        String uid;
        boolean exists;
        int attempts = 0;

        do {
            // Generate 9-character UID
            StringBuilder sb = new StringBuilder(9);
            for (int i = 0; i < 9; i++) {
                int index = random.nextInt(allowedCharacters.length());
                sb.append(allowedCharacters.charAt(index));
            }
            uid = sb.toString();

            // Check database
            exists = userRepository.existsByUserUid(uid);
            attempts++;

        } while (exists && attempts < 10); // Safety limit of 10 attempts

        if (exists) {
            throw new RuntimeException("Failed to generate unique User ID");
        }
        return uid;
    }
